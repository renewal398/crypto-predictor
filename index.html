Crypto Futures Multi-Indicator Suite (Final Enhanced)

//@version=5
indicator("Crypto Futures Multi-Indicator Suite (Final Enhanced)", overlay=true)

// === INPUTS ===
emaFastLen = input.int(50, title="EMA Fast (Short-term)")
emaSlowLen = input.int(200, title="EMA Slow (Long-term)")
rsiLen     = input.int(14, title="RSI Length")
stochLen   = input.int(14, title="Stoch RSI Length")
bbLen      = input.int(20, title="Bollinger Bands Length")
bbMult     = input.float(2.0, title="Bollinger Bands Multiplier")

// === EMAs ===
emaFast = ta.ema(close, emaFastLen)
emaSlow = ta.ema(close, emaSlowLen)

// === MACD ===
[macdLine, signalLine, _] = ta.macd(close, 12, 26, 9)
macdHist = macdLine - signalLine

// === RSI ===
rsi = ta.rsi(close, rsiLen)

// === True Stochastic RSI ===
rsi_src = rsi
rsiMin = ta.lowest(rsi_src, stochLen)
rsiMax = ta.highest(rsi_src, stochLen)
stochRSI = (rsi_src - rsiMin) / (rsiMax - rsiMin)
k = ta.sma(stochRSI, 3)
d = ta.sma(k, 3)

// === Bollinger Bands ===
basis = ta.sma(close, bbLen)
dev = bbMult * ta.stdev(close, bbLen)
upperBB = basis + dev
lowerBB = basis - dev

// === Volume Filter ===
volAvg = ta.sma(volume, 20)

// === Price Action Confirmation ===
bullishBreakout = close > high[1]
bearishBreakdown = close < low[1]

// === Buy/Sell Conditions ===
longCondition = close > emaFast and close > emaSlow and
    macdLine > signalLine and macdHist > 0 and
    rsi > 55 and k > d and close > basis and
    volume > volAvg and bullishBreakout

shortCondition = close < emaFast and close < emaSlow and
    macdLine < signalLine and macdHist < 0 and
    rsi < 45 and k < d and close < basis and
    volume > volAvg and bearishBreakdown

// === Plot EMAs and Bollinger Bands ===
plot(emaFast, title="EMA 50", color=color.orange)
plot(emaSlow, title="EMA 200", color=color.red)
plot(basis, title="BB Basis", color=color.blue)
plot(upperBB, title="BB Upper", color=color.green)
plot(lowerBB, title="BB Lower", color=color.green)

// === Plot Signals on Chart ===
plotshape(longCondition, title="Buy Signal", location=location.belowbar, color=color.lime, style=shape.labelup, text="BUY")
plotshape(shortCondition, title="Sell Signal", location=location.abovebar, color=color.red, style=shape.labeldown, text="SELL")

// === Background Highlighting ===
bgcolor(longCondition ? color.new(color.green, 85) : na)
bgcolor(shortCondition ? color.new(color.red, 85) : na)

// === MACD Histogram (Separate Pane) ===
plot(macdHist, title="MACD Histogram", color=color.purple, style=plot.style_columns)
hline(0, "Zero Line", color=color.gray)

// === Alert Conditions ===
alertcondition(longCondition, title="Buy Alert", message="Buy Signal Triggered")
alertcondition(shortCondition, title="Sell Alert", message="Sell Signal Triggered")

// === Label Annotations ===
if (longCondition)
    label.new(bar_index, low, text="BUY", style=label.style_label_up, color=color.green, textcolor=color.white)

if (shortCondition)
    label.new(bar_index, high, text="SELL", style=label.style_label_down, color=color.red, textcolor=color.white)
